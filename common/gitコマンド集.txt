■git コマンド

# 1. 変更があったことを確認
git status

# 2. すべての変更をステージング
git add .

# 3. 修正内容をメモ（コミット）
git commit -m "ハイブリッド・クラウド接続とセキュリティ自動化　初期構築その2"

# 4. GitHubに送信
git push origin main

#commitを取り消す
git reset --soft HEAD~1

git reset --soft f2efd37d71ee751e8efb05917a619d64c23873d1


■gitからファイルを消したい場合
git clone --mirror https://github.com/ユーザー名/リポジトリ.git
cd リポジトリ.git

git filter-repo --path <削除したいファイルのgitpath> --invert-paths
git push --force --all
git push --force --tags


#セキュリティ検知を有効にする
py -m pre_commit install

#確認コマンド
git config --list
